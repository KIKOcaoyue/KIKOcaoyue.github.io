**在此之前，我们先来回顾一下dfs序**

简单来说，dfs序就是dfs的顺序，如果大家学过先序遍历，那么dfs序大概就是先序遍历。由于dfs的特殊遍历性质，所以我们的树的dfs序都是连在一起的，大家可以自己画个图，可能就明白了。

**树链剖分是基于dfs序的基础上的**

##### Chapter 2

##### 一些函数介绍

1.**dfs1(x,u)**：

a.记录每个点的深度(dep[])

b.标记每个点的父节点(fa[])

c.标记含自己的子树的大小(sz[])

d.标记重儿子(son[])

2.**dfs2(x,topz)**:

a.给每个点标记一个新编号

b.给新节点赋值

c.处理每条链的top，(每条链的开头是哪一个)

**先处理重儿子，后处理轻儿子**





首先我们看一下dfs1()

```c++
void dfs1(int x,int u)
{
    dep[x] = dep[u]+1;
    fa[x] = u;
    sz[x] = 1;
    int mx = -1;
    for(int i=0;i<G[x].size();i++)
    {
        int v = G[x][i];
        if(v==u) continue;
        dfs1(v,x);
        sz[x] += sz[v];
        if(sz[v]>mx) 
        {
            son[x] = v;
            mx = sz[v];
        }
    }
}
```

其中：

dep[x]表示x的深度

sz[x]表示x的子树的大小(含x)

fa[x]表示x的父节点



然后我们看一下dfs2()

```c++
void dfs2(int x,int topz)
{
    id[x] = ++tot;
    nw[tot] = w[x];
    ltop[x] = topz;
    if(!son[x]) return;
    dfs2(son[x],topz);
    for(int i=0;i<G[x].size();i++)
    {
        int v = G[x][i];
        if(v==fa[x] || v==son[x]) continue;
        dfs2(v,v);
    }
}
```

其中：

id[x]表示节点x的新编号

nw[]表示新节点的值，w[]表示原节点的值

ltop[x]表示x所在链的首部

son[x]表示x的重儿子

fa[x]表示x的父节点



3.rangeupdate(x,y,z):

将节点x到节点y的路径上的点权都加上z

```c++
void rangeupdate(int x,int y,ll z)
{
    z %= p;
    while(ltop[x]!=ltop[y])
    {
        if(dep[ltop[x]]<dep[ltop[y]]) swap(x,y);
        BIT::add(id[ltop[x]],id[x],z);
        x = fa[ltop[x]];
    }
    if(dep[x]>dep[y]) swap(x,y);
    BIT::add(id[x],id[y],z);
}
```

具体表现为：

1.首先我们要将x,y调整到一条链上，如果他们目前不在一条链上，那么我们将节点深度较大的那个点所在的链更新，然后将该节点跳上链首的父节点。

![图片1.1](https://ss1.bdstatic.com/70cFuXSh_Q1YnxGkpoWK1HF6hhy/it/u=4015179012,4191848109&fm=11&gp=0.jpg)

**解析**：如果我们要把(4,5)的路径更新，显然通过肉眼观察，我们需要更新(4,3,2,5)这四个点。

现在5和(2,3,4)都不在一条链上，我们显然无法高效的维护他们的权值，所以我们先处理5这条链，(他现在是单独的一个点，但是我们还是认为他是一条链)，我们处理完5这条链之后，将它跳到5的父节点，也就是2这个节点，然后他现在就和(2,3,4)在一条链上了，我们就可以直接在线段树或者树状数组上维护他们的权值了。



4.treeupdate(int x,int z):将x及其子树上所有点的权值加上z

```c++
void treeupdate(int x,int z)
{
    z %= p;
    BIT::add(id[x],id[x]+sz[x]-1,z);
}
```

**解析**：由于树上的点是通过dfs来编号的，所以树上的每一个子树的编号都是连续的，由于我们事先记录好了每个子树的大小，所以对于每一个子树来说，他们的编号就是[id[x], id[x]+sz[x]-1]。



**此外，还有询问操作，他们和更新操作是一样的，这里就不赘述了**

### 小结

那么，为什么树链剖分可以高效的处理上述操作呢？

因为我们将树给划分开了，他们现在变成了一条一条链，并且他们的链的编号是连在一起的，所以我们可以很容易的使用线段树或者树状数组等高效的区间操作算法来解决他们。

```c++
#include <bits/stdc++.h>
using namespace std;
typedef long long ll;
const int maxn = 1e6+7;
int dep[maxn],fa[maxn],sz[maxn],son[maxn];
int id[maxn];
int w[maxn],nw[maxn];
int ltop[maxn];
vector<int> G[maxn];
ll sump[maxn],psum[maxn];
int tot = 0;
int n,m,r,p;
namespace BIT
{
    void add(int p,ll x)
    {
        for(int i=p;i<=tot;i+=(i & -i))
        {
            psum[i] += p*x;
            sump[i] += x;
        }
    }
    void add(int l,int r,ll x)
    {
        add(l,x);
        add(r+1,-x);
    }
    ll query(int p)
    {
        ll ans = 0;
        for(int i=p;i;i-=(i & -i))
        {
            ans += sump[i]*(p+1)-psum[i];
        }
        return ans;
    }
    ll query(int l,int r)
    {
        return (query(r)-query(l-1)+p)%p;
    }
    void build(int l,int r,int x)
    {
        add(l,r,x);
    }
}
namespace tcp
{
    void dfs1(int x,int u)
    {
        dep[x] = dep[u]+1;
        fa[x] = u;
        sz[x] = 1;
        int mx = -1;
        for(int i=0;i<G[x].size();i++)
        {
            int v = G[x][i];
            if(v==u) continue;
            dfs1(v,x);
            sz[x] += sz[v];
            if(sz[v]>mx)
            {
                son[x] = v;
                mx = sz[v];
            }
        }
    }
    void dfs2(int x,int topz)
    {
        id[x] = ++tot;
        nw[tot] = w[x];
        ltop[x] = topz;
        if(!son[x]) return;
        dfs2(son[x],topz);
        for(int i=0;i<G[x].size();i++)
        {
            int v = G[x][i];
            if(v==fa[x] || v==son[x]) continue;
            dfs2(v,v);
        }
    }
    void rangeupdate(int x,int y,ll z)
    {
        z %= p;
        while(ltop[x]!=ltop[y])
        {
            if(dep[ltop[x]]<dep[ltop[y]]) swap(x,y);
            BIT::add(id[ltop[x]],id[x],z);
            x = fa[ltop[x]];
        }
        if(dep[x]>dep[y]) swap(x,y);
        BIT::add(id[x],id[y],z);
    }
    void treeupdate(int x,int z)
    {
        z %= p;
        BIT::add(id[x],id[x]+sz[x]-1,z);
    }
    ll rangequery(int x,int y)
    {
        ll ans = 0;
        while(ltop[x]!=ltop[y])
        {
            if(dep[ltop[x]]<dep[ltop[y]]) swap(x,y);
            ans += BIT::query(id[ltop[x]],id[x]);
            ans %= p;
            x = fa[ltop[x]];
        }
        if(dep[x]>dep[y]) swap(x,y);
        ans += BIT::query(id[x],id[y]);
        ans %= p;
        return ans;
    }
    ll treequery(int x)
    {
        ll ans = 0;
        ans += BIT::query(id[x],id[x]+sz[x]-1);
        ans %= p;
        return ans;
    }
}
int main()
{
    scanf("%d%d%d%d",&n,&m,&r,&p);
    for(int i=1;i<=n;i++) scanf("%d",w+i);
    for(int i=0;i<n-1;i++)
    {
        int u,v;
        scanf("%d%d",&u,&v);
        G[u].push_back(v);
        G[v].push_back(u);
    }
    tcp::dfs1(r,0);
    tcp::dfs2(r,r);
    for(int i=1;i<=tot;i++) BIT::build(id[i],id[i],nw[id[i]]);
    for(int i=0;i<m;i++)
    {
        int op,x,y;
        ll z;
        scanf("%d",&op);
        if(op==1)
        {
            scanf("%d%d%lld",&x,&y,&z);
            tcp::rangeupdate(x,y,z);
        }
        else if(op==2)
        {
            scanf("%d%d",&x,&y);
            printf("%lld\n",tcp::rangequery(x,y));
        }
        else if(op==3)
        {
            scanf("%d%lld",&x,&z);
            tcp::treeupdate(x,z);
        }
        else
        {
            scanf("%d",&x);
            printf("%lld\n",tcp::treequery(x));
        }
    }
    return 0;
}
```
